114 // Scroll the entire screen if the cursor is beyond the last line
115 void cscroll() {
116  if(conio_cursor_y==CONIO_HEIGHT) {
117  if(conio_scroll_enable) {
118  // Scroll lines up
119  char* line1 = CONIO_SCREEN_TEXT;
120  char* line2 = CONIO_SCREEN_TEXT+CONIO_WIDTH;
121  for(char y=0;y<CONIO_HEIGHT-1;y++) {
122  ppuDataFetch(conio_cscroll_buffer, line2, CONIO_WIDTH);
123  ppuDataTransfer(line1, conio_cscroll_buffer, CONIO_WIDTH);
124  line1 += CONIO_WIDTH;
125  line2 += CONIO_WIDTH;
126  }
127  // Fill last line with space
128  ppuDataFill(CONIO_SCREEN_TEXT+CONIO_BYTES-CONIO_WIDTH, ' ', CONIO_WIDTH);
129  conio_line_text -= CONIO_WIDTH;
130  conio_cursor_y--;
131  } else {
132  gotoxy(0,0);
133  }
134  }
