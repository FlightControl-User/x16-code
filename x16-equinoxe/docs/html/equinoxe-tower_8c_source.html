142 void tower_animate()
143 {
144 
145  bank_push_set_bram(BRAM_ENGINE_TOWERS);
146 
147  for(unsigned char t=0; t<TOWERS_TOTAL; t++) {
148  if(towers.used[t] && towers.side[t] == SIDE_ENEMY) {
149 
150  switch(towers.state[t]) {
151  case 0:
152  unsigned char rnd = rand();
153  if(rnd<=5) towers.state[t] = 1;
154  break;
155  case 1:
156  towers.anim_start[t] = 0;
157  towers.anim_stop[t] = 11;
158  towers.anim_wait[t] = 0;
159  towers.state[t] = 2;
160  break;
161  case 2:
162  if(!towers.anim_wait[t]) {
163  towers.anim_wait[t] = towers.anim_speed[t];
164  if(towers.anim_state[t] >= towers.anim_stop[t]) {
165  towers.anim_wait[t] = 255;
166  towers.state[t] = 3;
167  } else {
168  towers.anim_state[t] += 1;
169  }
170  } else {
171  towers.anim_wait[t]--;
172  }
173  break;
174  case 3:
175  // Shoot
176  if(!towers.anim_wait[t]) {
177  towers.anim_wait[t] = 255;
178  towers.state[t] = 4;
179  } else {
180  towers.anim_wait[t]--;
181  }
182  break;
183  case 4:
184  if(!towers.anim_wait[t]) {
185  towers.anim_wait[t] = towers.anim_speed[t];
186  if(towers.anim_state[t] <= towers.anim_start[t]) {
187  towers.state[t] = 0;
188  } else {
189  towers.anim_state[t] -= 1;
190  }
191  }
192  towers.anim_wait[t]--;
193  break;
194  default:
195  }
196  }
197  }

199  bank_pull_bram();
