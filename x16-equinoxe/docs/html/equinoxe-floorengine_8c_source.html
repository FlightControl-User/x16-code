86 void floor_draw_row(unsigned char layer, floor_t* floor, unsigned char row, unsigned char column)
87 {
88  bank_push_set_bram(BRAM_FLOOR_CONTROL);

90  floor_parts_t* floor_parts = floor->floor_parts; // todo to pass this as the parameter!

94 
95  unsigned char shift = vera_layer0_get_rowshift();
96  mapbase_offset += ((word)row << shift);
97  mapbase_offset += column*8;
98 
99  unsigned char sr = ( (row % 4) / 2 ) * 2;
100  unsigned char r = (row % 2) * 2;
101 
102  unsigned char cache = FLOOR_CACHE(layer, row/4, column);
103  word segment = (word)floor_cache[cache];
104  segment = segment << 2;
105  vera_vram_data0_bank_offset(mapbase_bank, mapbase_offset,VERA_INC_1);
106  for(unsigned char sc=0;sc<2;sc++) {
107  unsigned char s = sc + sr;
108  unsigned char segment2 = floor->slab[(unsigned char)segment+s]; // todo i need to create a new variable because the optimizer deletes the old!
109  segment2 = segment2 << 2;
110  for(unsigned char c=0;c<2;c++) {
111  unsigned char tile = floor->composition[segment2+c+r]; // BRAM_FLOOR_CONTROL
112  unsigned int offset = floor_parts->floor_tile_offset[tile];
113  unsigned char palette = floor_parts->palette[tile];
114  palette = palette << 4;
115  *VERA_DATA0 = BYTE0(offset);
116  *VERA_DATA0 = palette | BYTE1(offset);
117  }
118  }
