110 void equinoxe_scrollfloor() {
111 
112 
113  // We only will execute the scroll logic when a scroll action needs to be done.
114  if(!game.screen_vscroll_wait--) {
115  game.screen_vscroll_wait = 4;
116 
117  floor_tile_row = (game.screen_vscroll-16) / 16;
118  floor_tile_row %= 32;
119 
120  // There are 16 scroll iterations as the height of the tiles is 16 pixels.
121  // Each segment is 4 tiles on the x axis, so the total amount of tiles are 64 tiles of 16 pixels wide.
122  // So there are 16 segments to be painted on each row.
123  // That allows to paint a segment per scroll action!
124  // We decrease the column for tiling, and ensure that we never go above 16.
125  floor_tile_column = (game.screen_vscroll-16) % 16;
126 
127  // We paint from bottom to top. Each paint segment is 64 pixels on the y axis, so we must paint every 4 rows.
128  // We paint when the row is the bottom row of the paint segment, so row 3. Row 0 is the top row of the segment.
129  if(floor_tile_row%4==3) {
130  #ifdef __TOWER
131  floor_paint(floor_tile_row/4, floor_tile_column);
132  tower_paint(floor_tile_row/4, floor_tile_column);
133  #else
134  floor_paint(floor_tile_row/4, floor_tile_column);
135  #endif
136  }
137 
138  // Now that the segment for the respective floor_tile_row and floor_tile_column has been painted,
139  // we can draw a cell from the painted segment. Note that when floor_tile_row is 0, 1 or 2,
140  // all paint segments will have been painted on the paint buffer, and the tiling will just pick
141  // row 2, 1 or 0 from the paint segment...
142  floor_draw_row(0, stage.floor, floor_tile_row, floor_tile_column);
143  #ifdef __LAYER1
144  floor_draw_row(1, stage.towers, floor_tile_row, floor_tile_column);
145  #endif
146 
147  #ifdef __TOWER
148  tower_logic();
149  #endif
150 
151  // Now we set the vertical scroll to the required scroll position.
152  vera_layer0_set_vertical_scroll(game.screen_vscroll);
153  #ifdef __LAYER1
154  vera_layer1_set_vertical_scroll(game.screen_vscroll);
155  #endif
156  game.screen_vscroll--;
157 
158  }
