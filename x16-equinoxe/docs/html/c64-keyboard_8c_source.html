56 // Read a single row of the keyboard matrix
57 // The row ID (0-7) of the keyboard matrix row to read. See the C64 key matrix for row IDs.
58 // Returns the keys pressed on the row as bits according to the C64 key matrix.
59 // Notice: If the C64 normal interrupt is still running it will occasionally interrupt right between the read & write
60 // leading to erroneous readings. You must disable the normal interrupt or sei/cli around calls to the keyboard matrix reader.
61 char keyboard_matrix_read(char rowid) {
62  CIA1->PORT_A = keyboard_matrix_row_bitmask[rowid];
63  char row_pressed_bits = ~CIA1->PORT_B;
64  return row_pressed_bits;
65 }

67 // Determines whether a specific key is currently pressed by accessing the matrix directly
68 // The key is a keyboard code defined from the keyboard matrix by %00rrrccc, where rrr is the row ID (0-7) and ccc is the column ID (0-7)
69 // All keys exist as as KEY_XXX constants.
70 // Returns zero if the key is not pressed and a non-zero value if the key is currently pressed
71 char keyboard_key_pressed(char key) {
72  char colidx = key&7;
73  char rowidx = key>>3;
74  return keyboard_matrix_read(rowidx) & keyboard_matrix_col_bitmask[colidx];
75 }

99 // Any shift is pressed

101 // The values scanned values for each row. Set by keyboard_scan() and used by keyboard_get_event()

104 // Scans the entire matrix to determine which keys have been pressed/depressed.
105 // Generates keyboard events into the event buffer. Events can be read using keyboard_event_get().
106 // Handles debounce and only generates events when the status of a key changes.
107 // Also stores current status of modifiers in keyboard_modifiers.
108 void keyboard_event_scan() {
109  char keycode = 0;
110  for(char row : 0..7) {
111  char row_scan = keyboard_matrix_read(row);
112  if(row_scan!=keyboard_scan_values[row]) {
113  // Something has changed on the keyboard row - check each column
114  for(char col : 0..7){
115  // XOR of row scan with the last seen row scan AND'ed with the col bitmask will be non-0 if the key status is changed
116  if(((row_scan^keyboard_scan_values[row])&keyboard_matrix_col_bitmask[col])!=0) {
117  // Key(row, col) status has changed. We have an event.
118  // Only process event if there is still room in the buffer
119  if(keyboard_events_size!=8) {
120  // AND of row scan and bit mask determines if key is pressed or released
121  char event_type = row_scan&keyboard_matrix_col_bitmask[col];
122  if(event_type==0) {
123  // Key released
124  keyboard_events[keyboard_events_size++] = keycode|$40;
125  } else {
126  // Key pressed
127  keyboard_events[keyboard_events_size++] = keycode;
128  }
129  }
130  }
131  keycode++;
132  }
133  // Store the current keyboard status for the row to debounce
134  keyboard_scan_values[row] = row_scan;
135  } else {
136  // Update current keycode
137  keycode = keycode + 8;
138  }
139  }
140  // Update the keyboard modifiers
141  keyboard_modifiers = 0;
142  if(keyboard_event_pressed(KEY_LSHIFT)!= 0) {
143  keyboard_modifiers = keyboard_modifiers | KEY_MODIFIER_LSHIFT;
144  }
145  if(keyboard_event_pressed(KEY_RSHIFT)!= 0) {
146  keyboard_modifiers = keyboard_modifiers | KEY_MODIFIER_RSHIFT;
147  }
148  if(keyboard_event_pressed(KEY_CTRL)!= 0) {
149  keyboard_modifiers = keyboard_modifiers | KEY_MODIFIER_CTRL;
150  }
151  if(keyboard_event_pressed(KEY_COMMODORE)!= 0) {
152  keyboard_modifiers = keyboard_modifiers | KEY_MODIFIER_COMMODORE;
153  }
154 }

156 // Determine if a specific key is currently pressed based on the last keyboard_event_scan()
157 // Returns 0 is not pressed and non-0 if pressed
158 char keyboard_event_pressed(char keycode) {
159  char row_bits = keyboard_scan_values[keycode>>3];
160  return row_bits & keyboard_matrix_col_bitmask[keycode&7];
161 }

163 // Get the next event from the keyboard event buffer.
164 // Returns $ff if there is no event waiting. As all events are <$7f it is enough to examine bit 7 when determining if there is any event to process.
165 // The buffer is filled by keyboard_event_scan()
166 char keyboard_event_get() {
167  if(keyboard_events_size==0) {
168  return $ff;
169  } else {
170  return keyboard_events[--keyboard_events_size];
171  }
172 }
