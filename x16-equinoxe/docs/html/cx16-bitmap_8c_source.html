65 void bitmap_init(unsigned char layer, unsigned char bank, unsigned int offset)
66 {
67  __bitmap.address = MAKELONG(offset, bank);
68  __bitmap.layer = layer;
69  if(layer) {
70  __bitmap.color_depth = (*VERA_L1_CONFIG & VERA_LAYER_COLOR_DEPTH_MASK);
71  } else {
72  __bitmap.color_depth = (*VERA_L0_CONFIG & VERA_LAYER_COLOR_DEPTH_MASK);
73  }
74  __bitmap.hscale = bitmap_hscale(); // Returns 1 when 640 and 2 when 320, 3 when 160.
75  __bitmap.vscale = bitmap_vscale(); // Returns 1 when 480 and 2 when 240, 3 when 160.
76 
77  unsigned char bitmask = bitmasks[__bitmap.color_depth];
78  signed char bitshift = bitshifts[__bitmap.color_depth];
79 
80  for(unsigned int x=0; x<630; x++) {
81  // 1 BPP
82  if(__bitmap.color_depth==0) {
83  __bitmap.plot_x[x] = (x >> 3);
84  __bitmap.plot_bitmask[x] = bitmask;
85  __bitmap.plot_bitshift[x] = (unsigned char)bitshift;
86  bitshift -= 1;
87  bitmask >>= 1;
88  }
89  // 2 BPP
90  if(__bitmap.color_depth==1) {
91  __bitmap.plot_x[x] = (x >> 2);
92  __bitmap.plot_bitmask[x] = bitmask;
93  __bitmap.plot_bitshift[x] = (unsigned char)bitshift;
94  bitshift -= 2;
95  bitmask >>= 2;
96  }
97  // 4 BPP
98  if(__bitmap.color_depth==2) {
99  __bitmap.plot_x[x] = (x >> 1);
100  __bitmap.plot_bitmask[x] = bitmask;
101  __bitmap.plot_bitshift[x] = (unsigned char)bitshift;
102  bitshift -= 4;
103  bitmask >>= 4;
104  }
105  // 8 BPP
106  if(__bitmap.color_depth==3) {
107  __bitmap.plot_x[x] = x;
108  __bitmap.plot_bitmask[x] = bitmask;
109  __bitmap.plot_bitshift[x] = (unsigned char)bitshift;
110  }
111  if(bitshift<0) {
112  bitshift = bitshifts[__bitmap.color_depth];
113  }
114  if(bitmask==0) {
115  bitmask = bitmasks[__bitmap.color_depth];
116  }
117  }
118 
119  // This sets the right delta to skip a whole line based on the scale, depending on the color depth.
120  unsigned int hdelta = hdeltas[(__bitmap.color_depth<<2)+__bitmap.hscale];
121  // We start at the bitmap offset; The plot_y contains the bitmap offset embedded so we know where a line starts.
122  unsigned long yoffs = __bitmap.address;
123  for(unsigned int y=0; y<479; y++) {
124  __bitmap.plot_y[y] = yoffs;
125  yoffs = yoffs + hdelta;
126  }
127 }

141 void bitmap_plot(unsigned int x, unsigned int y, unsigned char c) {
142  // Needs unsigned int arrays arranged as two underlying char arrays to allow char* plotter_x = plot_x[x]; - and eventually - char* plotter = plot_x[x] + plot_y[y];
143  unsigned long plot_x = __bitmap.plot_x[x];
144  unsigned long plot_y = __bitmap.plot_y[y];
145  unsigned long plotter = plot_x+plot_y;
146  unsigned char bitshift = __bitmap.plot_bitshift[x];
147  c = bitshift?c<<(bitshift):c;
148  vera_vram_data0_address(plotter,VERA_INC_0);
149  *VERA_DATA0 = (*VERA_DATA0 & ~__bitmap.plot_bitmask[x]) | c;
