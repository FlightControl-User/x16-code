186 unsigned char cbm_k_load(char* address, char verify)
187 {
188  unsigned char status;
189  asm {
190  //LOAD. Load or verify file. (Must call SETLFS and SETNAM beforehands.)
191  // Input: A: 0 = Load, 1-255 = Verify; X/Y = Load address (if secondary address = 0).
192  // Output: Carry: 0 = No errors, 1 = Error; A = KERNAL error code (if Carry = 1); X/Y = Address of last byte loaded/verified (if Carry = 0).
193  .byte $db
194  ldx address
195  ldy address+1
196  lda verify
197  jsr $ffd5
198  bcs !error+
199  lda #$ff
200  !error:
201  sta status
202  }
203  return status;
204 }
