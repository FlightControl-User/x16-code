1 // A flexible sprite multiplexer routine for 32 sprites.
2 // Usage:
3 // - Once:
4 // - plexInit(screen): Initialize the data structures and set the screen address
5 // Each frame:
6 // - Set x-pos, y-pos and pointer in PLEX_XPOS[id], PLEX_YPOS[id], PLEX_PTR[id]
7 // - plexSort() Sorts the sprites according to y-positions and prepares for showing them. This uses an insertion sort that is quite fast when the relative order of the sprites does not change very much.
8 // - plexShowSprite() Shows the next sprite by copying values from PLEX_XXX[] to an actual sprite. Actual sprites are used round-robin. This should be called once for each of the 24 virtual sprites.
9 // - plexFreeNextYpos() Returns the Y-position where the next sprite is available to be shown (ie. the next pos where the next sprite is no longer in use showing something else).
10 // - plexShowNextYpos() Returns the Y-position of the next sprite to show.
11 //
12 // In practice a good method is to wait until the raster is beyond plexFreeNextYpos() and then call plexShowSprite(). Repeat until all 32 sprites have been shown.
13 // TODO: Let the caller specify the number of sprites to use (or add PLEX_ENABLE[PLEX_COUNT])

58 // Ensure that the indices in PLEX_SORTED_IDX is sorted based on the y-positions in PLEX_YPOS
59 // Assumes that the positions are nearly sorted already (as each sprite just moves a bit)
60 // Uses an insertion sort:
61 // 1. Moves a marker (m) from the start to end of the array. Every time the marker moves forward all elements before the marker are sorted correctly.
62 // 2a. If the next element after the marker is larger that the current element
63 // the marker can be moved forwards (as the sorting is correct).
64 // 2b. If the next element after the marker is smaller than the current element:
65 // elements before the marker are shifted right one at a time until encountering one smaller than the current one.
66 // It is then inserted at the spot. Now the marker can move forward.
67 void plexSort() {
68  for(char m: 0..PLEX_COUNT-2) {
69  char nxt_idx = PLEX_SORTED_IDX[m+1];
70  char nxt_y = PLEX_YPOS[nxt_idx];
71  if(nxt_y<PLEX_YPOS[PLEX_SORTED_IDX[m]]) {
72  // Shift values until we encounter a value smaller than nxt_y
73  char s = m;
74  do {
75  PLEX_SORTED_IDX[s+1] = PLEX_SORTED_IDX[s];
76  s--;
77  } while((s!=0xff) && (nxt_y<PLEX_YPOS[PLEX_SORTED_IDX[s]]));
78  // store the mark at the found position
79  s++;
80  PLEX_SORTED_IDX[s] = nxt_idx;
81  }
82  }
83  // Prepare for showing the sprites
84  plex_show_idx = 0;
85  plex_sprite_idx = 0;
86  plex_sprite_msb = 1;
87  plexFreePrepare();
88 }

92 void plexShowSprite() {
93  char plex_sprite_idx2 = plex_sprite_idx*2;
94  char ypos = PLEX_YPOS[PLEX_SORTED_IDX[plex_show_idx]];
95  SPRITES_YPOS[plex_sprite_idx2] = ypos;
96  plexFreeAdd(ypos);
97  PLEX_SCREEN_PTR[plex_sprite_idx] = PLEX_PTR[PLEX_SORTED_IDX[plex_show_idx]];
98  char xpos_idx = PLEX_SORTED_IDX[plex_show_idx];
99  SPRITES_XPOS[plex_sprite_idx2] = (char)PLEX_XPOS[xpos_idx];
100  if(BYTE1(PLEX_XPOS[xpos_idx])!=0) {
101  *SPRITES_XMSB |= plex_sprite_msb;
102  } else {
103  *SPRITES_XMSB &= (0xff^plex_sprite_msb);
104  }
105  plex_sprite_idx = (plex_sprite_idx+1)&7;
106  plex_show_idx++;
107  plex_sprite_msb <<=1;
108  if(plex_sprite_msb==0) {
109  plex_sprite_msb = 1;
110  }
111 }
