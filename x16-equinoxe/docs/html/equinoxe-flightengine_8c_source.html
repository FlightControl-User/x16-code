110 vera_sprite_image_offset sprite_image_cache_vram(fe_sprite_index_t fe_sprite_index, unsigned char fe_sprite_image_index) {
111  // check if the image in vram is in use where the fe_sprite_vram_image_index is pointing to.
112  // if this vram_image_used is false, that means that the image in vram is not in use anymore (not displayed or destroyed).
113 
114  unsigned int image_index = sprite_cache.offset[fe_sprite_index] + fe_sprite_image_index;
115 
116  // We retrieve the image from BRAM from the sprite_control bank.
117  // TODO: what if there are more sprite control data than that can fit into one CX16 bank?
118  bank_push_set_bram(BRAM_SPRITE_CONTROL);
119  heap_bram_fb_handle_t handle_bram = sprite_bram_handles[image_index];
120  bank_pull_bram();
121 
122  // We declare temporary variables for the vram memory handles.
123  lru_cache_data_t vram_handle;
124  vram_bank_t vram_bank;
125  vram_offset_t vram_offset;
126 
127  // We check if there is a cache hit?
128  lru_cache_index_t vram_index = lru_cache_index(&sprite_cache_vram, image_index);
129  lru_cache_data_t lru_cache_data;
130  vera_sprite_image_offset sprite_offset;
131  if (vram_index != 0xFF) {
132 
133  // So we have a cache hit, so we can re-use the same image from the cache and we win time!
134  vram_handle = lru_cache_get(&sprite_cache_vram, vram_index);
135 
136  // Now that we are sure that there is sufficient space in vram and on the cache, we allocate a new element.
137  // Dynamic allocation of sprites in vera vram.
138  vram_bank = vera_heap_data_get_bank(VERA_HEAP_SEGMENT_SPRITES, vram_handle);
139  vram_offset = vera_heap_data_get_offset(VERA_HEAP_SEGMENT_SPRITES, vram_handle);
140 
141  sprite_offset = vera_sprite_get_image_offset(vram_bank, vram_offset);
142  } else {
143 
144  // The idea of this section is to free up lru_cache and/or vram memory until there is sufficient space available.
145  // The size requested contains the required size to be allocated on vram.
146  vera_heap_size_int_t vram_size_required = sprite_cache.size[fe_sprite_index];
147 
148  // We check if the vram heap has sufficient memory available for the size requested.
149  // We also check if the lru cache has sufficient elements left to contain the new sprite image.
150  bool vram_has_free = vera_heap_has_free(VERA_HEAP_SEGMENT_SPRITES, vram_size_required);
151  bool lru_cache_not_free = lru_cache_max(&sprite_cache_vram);
152 
153  // Free up the lru_cache and vram memory until the requested size is available!
154  // This ensures that vram has sufficient place to allocate the new sprite image.
155  while(lru_cache_not_free || !vram_has_free) {
156 
157  // If the cache is at it's maximum, before we can add a new element, we must remove the least used image.
158  // We search for the least used image in vram.
159  lru_cache_key_t vram_last = lru_cache_last(&sprite_cache_vram);
160 
161  // We delete the least used image from the vram cache, and this function returns the stored vram handle obtained by the vram heap manager.
162  vram_handle = lru_cache_delete(&sprite_cache_vram, vram_last);
163  if(vram_handle==0xFFFF) {
164  #ifdef __INCLUDE_PRINT
165  gotoxy(0,59);
166  printf("error! vram_handle is nothing!");
167  #endif
168  }
169 
170  // And we free the vram heap with the vram handle that we received.
171  // But before we can free the heap, we must first convert back from teh sprite offset to the vram address.
172  // And then to a valid vram handle :-).
173  vera_heap_free(VERA_HEAP_SEGMENT_SPRITES, vram_handle);
174  vram_has_free = vera_heap_has_free(VERA_HEAP_SEGMENT_SPRITES, vram_size_required);
175  }
176 
177  // Now that we are sure that there is sufficient space in vram and on the cache, we allocate a new element.
178  // Dynamic allocation of sprites in vera vram.
179  vram_handle = vera_heap_alloc(VERA_HEAP_SEGMENT_SPRITES, (unsigned long)sprite_cache.size[fe_sprite_index]);
180  vram_bank = vera_heap_data_get_bank(VERA_HEAP_SEGMENT_SPRITES, vram_handle);
181  vram_offset = vera_heap_data_get_offset(VERA_HEAP_SEGMENT_SPRITES, vram_handle);
182 
183 #ifdef __CPULINES
184  vera_display_set_border_color(GREEN);
185 #endif
186  memcpy_vram_bram(vram_bank, vram_offset, heap_bram_fb_bank_get(handle_bram), (bram_ptr_t)heap_bram_fb_ptr_get(handle_bram), sprite_cache.size[fe_sprite_index]);
187 #ifdef __CPULINES
188  vera_display_set_border_color(BLACK);
189 #endif
190  sprite_offset = vera_sprite_get_image_offset(vram_bank, vram_offset);
191  lru_cache_insert(&sprite_cache_vram, image_index, vram_handle);
192  }
193 
194  // We return the image offset in vram of the sprite to be drawn.
195  // This offset is used by the vera image set offset function to directly change the image displayed of the sprite!
196  return sprite_offset;
197 }
