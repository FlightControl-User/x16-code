20  signed int xi = (x>=0)?x:-x;

22  for( char i: 0..CORDIC_ITERATIONS_16-1) {
23  if(yi==0) {
24  // We found the correct angle!
25  break;
26  }
27  // Optimized shift of 2 values: xd=xi>>i; yd=yi>>i
28  signed int xd = xi;
29  signed int yd = yi;
30  char shift = i;
31  while(shift>=2) {
32  xd >>= 2;
33  yd >>= 2;
34  shift -=2;
35  }
36  if(shift) {
37  xd >>= 1;
38  yd >>= 1;
39  }
40  if(yi>=0) {
41  xi += yd;

43  angle += CORDIC_ATAN2_ANGLES_16[i];
44  } else {

49  }
