122 // Return the current screen size.
123 void screensize(unsigned char* x, unsigned char* y) {
124  // VERA returns in VERA_DC_HSCALE the value of 128 when 80 columns is used in text mode,
125  // and the value of 64 when 40 columns is used in text mode.
126  // Basically, 40 columns mode in the VERA is a double scan mode.
127  // Same for the VERA_DC_VSCALE mode, but then the subdivision is 60 or 30 rows.
128  // I still need to test the other modes, but this will suffice for now for the pure text modes.
129  char hscale = (*VERA_DC_HSCALE) >> 7;
130  *x = (40 << hscale)-1;
131  char vscale = (*VERA_DC_VSCALE) >> 7;
132  *y = (30 << vscale)-1;
133  //printf("%u, %u\n", *x, *y);
134 }

325 void screenlayer(char layer, char mapbase, char config)
326 {
327  unsigned char const VERA_LAYER_DIM[4] = {31, 63, 127, 255};
328  unsigned int const VERA_LAYER_SKIP[4] = {64, 128, 256, 512};
329 
330  __mem char vera_dc_hscale_temp = *VERA_DC_HSCALE;
331  __mem char vera_dc_vscale_temp = *VERA_DC_VSCALE;
332 
333  __conio.layer = 0;
334  __conio.mapbase_bank = mapbase >> 7;
335  __conio.mapbase_offset = MAKEWORD((mapbase)<<1,0);
336  __conio.mapwidth = VERA_LAYER_DIM[ (config & VERA_LAYER_WIDTH_MASK) >> 4];
337  __conio.mapheight = VERA_LAYER_DIM[ (config & VERA_LAYER_HEIGHT_MASK) >> 6];
338  // __conio.rowshift = ((config & VERA_LAYER_WIDTH_MASK)>>4)+6;
339  __conio.rowskip = VERA_LAYER_SKIP[(config & VERA_LAYER_WIDTH_MASK)>>4];
340  __conio.width = (40 << (char)(vera_dc_hscale_temp == 0x80))-1;
341  __conio.height = (30 << (char)(vera_dc_vscale_temp == 0x80))-1;
342 
343  unsigned int mapbase_offset = __conio.mapbase_offset;
344  for(register char y=0; y<__conio.height; y++) {
345  __conio.offsets[y] = mapbase_offset;
346  mapbase_offset += __conio.rowskip;
347  }
348 }
