11 struct F018_DMAGIC {
12  // $D700 ADDRLSBTRIG DMAgic DMA list address LSB, and trigger DMA (when written).
13  // We also clear out the upper address bits in case an enhanced job had set them.
14  char ADDRLSBTRIG;
15  // $D701 ADDRMSB DMA list address high byte (address bits 8 -- 15).
16  char ADDRMSB;
17  // $D702 ADDRBANK DMA list address bank (address bits 16 -- 22). Writing clears $D704.
18  char ADDRBANK;
19  // $D703 EN018B DMA enable F018B mode (adds sub-command byte )
20  // bit 0 enable F018B mode.
21  char EN018B;
22  // $D704 ADDRMB DMA list address mega-byte
23  char ADDRMB;
24  // $D705 ETRIG Set low-order byte of DMA list address, and trigger Enhanced DMA job
25  // Works like $D700, but enables DMA option lists.
26  char ETRIG;
27  // $D706-$D70D Unused
28  char UNUSED1[8];
29  // $D70E ADDRLSB DMA list address low byte (address bits 0 -- 7) WITHOUT STARTING A DMA JOB
30  // (used by Hypervisor for unfreezing DMA-using tasks)
31  char ADDRLSB;
32  // $D70F Unused
33  char UNUSED2;
34  // $D710 MISC (non-DMA) options
35  // $D710.0 - MISC:BADLEN Enable badline emulation
36  // $D710.1 - MISC:SLIEN Enable 6502-style slow (7 cycle) interrupts
37  // $D710.2 - MISC:VDCSEN Enable VDC interface simulation
38  char MISC;
