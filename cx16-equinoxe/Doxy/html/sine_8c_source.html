70 // x: unsigned long input u[4.28] in the interval $00000000 - PI2_u4f28
71 // result: signed int sin(x) s[0.15] - using the full range -$7fff - $7fff
72 signed int sin16s(unsigned long x) {
73  // Move x1 into the range 0-PI/2 using sine mirror symmetries
74  char isUpper = 0;
75  if(x >= PI_u4f28 ) {
76  x = x - PI_u4f28;
77  isUpper = 1;
78  }
79  if(x >= PI_HALF_u4f28 ) {
80  x = PI_u4f28 - x;
81  }
82  // sinx = x - x^3/6 + x5/128;
83  unsigned int x1 = >x<<3; // u[1.15]
84  unsigned int x2 = mulu16_sel(x1, x1, 0); // u[2.14] x^2
85  unsigned int x3 = mulu16_sel(x2, x1, 1); // u[2.14] x^3
86  unsigned int x3_6 = mulu16_sel(x3, $10000/6, 1); // u[1.15] x^3/6;
87  unsigned int usinx = x1 - x3_6; // u[1.15] x - x^3/6
88  unsigned int x4 = mulu16_sel(x3, x1, 0); // u[3.13] x^4
89  unsigned int x5 = mulu16_sel(x4, x1, 0); // u[4.12] x^5
90  unsigned int x5_128 = x5>>4; // // u[1.15] x^5/128 -- much more efficient than mul_u16_sel(x5, $10000/128, 3);
91  usinx = usinx + x5_128; // u[1.15] (first bit is always zero)
92  signed int sinx = (signed int)usinx; // s[0.15]
93  if(isUpper!=0) {
