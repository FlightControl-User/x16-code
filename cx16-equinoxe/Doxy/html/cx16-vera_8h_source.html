26 // To access the VRAM (which is 128kB in size) an indirection mechanism is used.
27 // First the address to be accessed needs to be set (ADDRx_L/ADDRx_M/ADDRx_H) and
28 // then the data on that VRAM address can be read from or written to via the DATA0/1 register.
29 // To make accessing the VRAM more efficient an auto-increment mechanism is present.
30 // These 3 registers are multiplexed using the ADDR_SEL in the CTRL register.
31 // When ADDR_SEL = 0, ADDRx_L/ADDRx_M/ADDRx_H become ADDR0_L/ADDR0_M/ADDR0_H.
32 // When ADDR_SEL = 1, ADDRx_L/ADDRx_M/ADDRx_H become ADDR1_L/ADDR1_M/ADDR1_H.
33 // See https://github.com/commanderx16/x16-emulator/wiki/(VERA-0.8)-Registers-$9F23-and-$9F24-(and-$9F25)

249 // The VERA structure of a sprite (8 bytes)
250 // 128*8 bytes located at $1FC00-$1FFFF
251 struct VERA_SPRITE {
252  // 0-1 ADDRESS Address and mode
253  // - bits 0-11: Address (16:5)
254  // - bits 15: Mode (0:4 bpp 1:8 bpp)
255  unsigned int ADDR;
256  // 2-3 X (9:0)
257  unsigned int X;
258  // 4-5 Y (9:0)
259  unsigned int Y;
260  // 6 CTRL1 Control 1
261  // - bits 4-7 Collision mask
262  // - bits 2-3 Z-depth ( 0:Sprite disabled, 1:Sprite between background and layer 0, 2:Sprite between layer 0 and layer 1, 3:Sprite in front of layer 1 )
263  // - bits 1 V-flip
264  // - bits 0 H-flip
265  char CTRL1;
266  // 7 CTRL2 Control 2
267  // - bits 6-7 Sprite height ( 0:8 pixels, 1:16 pixels, 2:32 pixels, 3:64 pixels )
268  // - bits 4-5 Sprite width ( 0:8 pixels, 1:16 pixels, 2:32 pixels, 3:64 pixels )
269  // - bits 0-3 Palette offset (if 4bpp Color index 1-15 is modified by adding 16 x palette offset)
270  char CTRL2;
271 };
