29 // Output one character at the current cursor position
30 // Moves the cursor forward. Scrolls the entire screen if needed
31 void cputc(char c) {
32  char screenCode;
33  if (c == '\r') { // 0x0d, CR = just set the cursor x value to 0
34  *COLCRS = 0;
35  setcursor();
36  } else if(c == '\n' || c == 0x9b) { // 0x0a LF, or atascii EOL
37  *COLCRS = 0;

81 void newline() {
82  if ((*ROWCRS)++ == CONIO_HEIGHT) {
83  if (conio_scroll_enable == 1) {
84  // first hide the current screen cursor, so it doesn't scroll, but only if it is on
85  if (conio_display_cursor == 1) {
86  **OLDADR ^= 0x80;
87  }
88  // move screen up 1 line
89  char * start = *SAVMSC;
90  memcpy(start, start + CONIO_WIDTH, CONIO_WIDTH * 23);
91  memset(start + CONIO_WIDTH * 23, 0x00, CONIO_WIDTH);
92  *ROWCRS = CONIO_HEIGHT - 1;
93  } else {
94  gotoxy(0, 0);
95  }
96  }
97  setcursor();
98 }

100 // Handles cursor movement, displaying it if required, and inverting character it is over if there is one (and enabled)
101 void setcursor() {
102  // save the current oldchr into oldadr
103  **OLDADR = *OLDCHR;
104  // work out the new location for oldadr based on new column/row
105  char * loc = cursorLocation();
106  char c = *loc;
107  *OLDCHR = c;
108  *OLDADR = loc;
109  if (conio_display_cursor == 0) {
110  // cursor is off, don't affect the character at all, but do turn off cursor
111  *CRSINH = 1;
112  } else {
113  // cursor is on, so invert the inverse bit and turn cursor on
114  *CRSINH = 0;
115  c = c ^ 0x80;
116  }
117  **OLDADR = c;
118 }

231 // create a static table to map char value to screen value
232 // use KickAsm functions to create a table of code -> screen code values, using cc65 algorithm
233 char rawmap[0x100] = kickasm {{
234  .var ht = Hashtable().put(0,64, 1,0, 2,32, 3,96) // the table for converting bit 6,7 into ora value
235  .for(var i=0; i<256; i++) {
236  .var idx = (i & $60) / 32
237  .var mask = i & $9f
238  .byte mask | ht.get(idx)
239  }
240 }};
