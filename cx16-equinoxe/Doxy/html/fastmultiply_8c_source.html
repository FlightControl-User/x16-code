17 // Initialize the mulf_sqr multiplication tables with f(x)=int(x*x/4)
18 void mulf_init() {
19  // Fill mulf_sqr1 = f(x) = int(x*x/4): If f(x) = x*x/4 then f(x+1) = f(x) + x/2 + 1/4
20  unsigned int sqr = 0; // sqr = (x*x)/4
21  char x_2 = 0; // x/2
22  char c = 0; // Counter used for determining x%2==0
23  char* sqr1_hi = mulf_sqr1_hi+1;
24  for(char* sqr1_lo = mulf_sqr1_lo+1; sqr1_lo!=mulf_sqr1_lo+512; sqr1_lo++) {
25  if((++c&1)==0) {
26  x_2++; // increase i/2 on even numbers
27  sqr++; // sqr++ on even numbers because 1 = 2*1/4 (from the two previous numbers) + 1/2 (half of the previous uneven number)
28  }
29  *sqr1_lo = <sqr;
30  *sqr1_hi++ = >sqr;
31  sqr = sqr + x_2; // sqr = sqr + i/2 (when uneven the 1/2 is not added here - see above)
32  }
33  // Fill mulf_sqr2 = g(x) = f(x-255) : If x-255<0 then g(x)=f(255-x) (because x*x = -x*-x)
34  // g(0) = f(255), g(1) = f(254), ..., g(254) = f(1), g(255) = f(0), g(256) = f(1), ..., g(510) = f(255), g(511) = f(256)
35  char x_255 = (char)-1; //Start with g(0)=f(255)
36  char dir = $ff; // Decrease or increase x_255 - initially we decrease
37  char* sqr2_hi = mulf_sqr2_hi;
38  for(char* sqr2_lo = mulf_sqr2_lo; sqr2_lo!=mulf_sqr2_lo+511; sqr2_lo++) {
39  *sqr2_lo = mulf_sqr1_lo[x_255];
40  *sqr2_hi++ = mulf_sqr1_hi[x_255];
41  x_255 = x_255 + dir;
42  if(x_255==0) {
43  dir = 1; // when x_255=0 then start counting up
44  }
45  }
46  // Set the very last value g(511) = f(256)
47  *(mulf_sqr2_lo+511) = *(mulf_sqr1_lo+256);
48  *(mulf_sqr2_hi+511) = *(mulf_sqr1_hi+256);
49 }
