9 // The text screen base address, which is a 16:0 bit value in VERA VRAM.
10 // That is 128KB addressable space, thus 17 bits in total.
11 // CONIO_SCREEN_TEXT contains bits 15:0 of the address.
12 // CONIO_SCREEN_BANK contains bit 16, the the 64K memory bank in VERA VRAM (the upper 17th bit).
13 // !!! note that these values are not const for the cx16!
14 // This conio implements the two layers of VERA, which can be layer 0 or layer 1.
15 // Configuring conio to output to a different layer, will change these fields to the address base
16 // configured using VERA_L0_MAPBASE = 0x9f2e or VERA_L1_MAPBASE = 0x9f35.
17 // Using the function setscreenlayer(layer) will re-calculate using CONIO_SCREEN_TEXT and CONIO_SCREEN_BASE
18 // based on the values of VERA_L0_MAPBASE or VERA_L1_MAPBASE, mapping the base address of the selected layer.
19 // The function setscreenlayermapbase(layer,mapbase) allows to configure bit 16:9 of the
20 // mapbase address of the time map in VRAM of the selected layer VERA_L0_MAPBASE or VERA_L1_MAPBASE.

87 unsigned char kbhit(void) {
88 
89  char ch = 0;
90  char* const chptr = &ch;
91 
92  char* const IN_DEV = (char*)$028A; // Current input device number
93  char* const GETIN = (char*)$FFE4; // CBM GETIN API
94 
95  kickasm(uses chptr, uses IN_DEV, uses GETIN) {{
96 
97  jsr _kbhit
98  bne L3
99 
100  jmp continue1
101 
102  .var via1 = $9f60 //VIA#1
103  .var d1pra = via1+1
104 
105  _kbhit:
106  ldy d1pra // The count of keys pressed is stored in RAM bank 0.
107  stz d1pra // Set d1pra to zero to access RAM bank 0.
108  lda $A00A // Get number of characters from this address in the ROM of the CX16 (ROM 38).
109  sty d1pra // Set d1pra to previous value.
110  rts
111 
112  L3:
113  ldy IN_DEV // Save current input device
114  stz IN_DEV // Keyboard
115  phy
116  jsr GETIN // Read char, and return in .A
117  ply
118  sta chptr // Store the character read in ch
119  sty IN_DEV // Restore input device
120  ldx #>$0000
121  rts
122 
123  continue1:
124  nop
125  }}
126 
127  return ch;
128 }

166 void screensize(unsigned byte* x, unsigned byte* y) {
167  // VERA returns in VERA_DC_HSCALE the value of 128 when 80 columns is used in text mode,
168  // and the value of 64 when 40 columns is used in text mode.
169  // Basically, 40 columns mode in the VERA is a double scan mode.
170  // Same for the VERA_DC_VSCALE mode, but then the subdivision is 60 or 30 rows.
171  // I still need to test the other modes, but this will suffice for now for the pure text modes.
172  char hscale = (*VERA_DC_HSCALE) >> 7;
173  *x = 40 << hscale;
174  char vscale = (*VERA_DC_VSCALE) >> 7;
175  *y = 30 << vscale;
176  //printf("%u, %u\n", *x, *y);
177 }

206 
207  conio_addr += conio_cursor_x[cx16_conio.conio_screen_layer] << 1;
208  if(c=='\n') {

210  } else {
211  // Select DATA0
212  *VERA_CTRL &= ~VERA_ADDRSEL;
213  // Set address
214  *VERA_ADDRX_L = <conio_addr;
215  *VERA_ADDRX_M = >conio_addr;
216  *VERA_ADDRX_H = cx16_conio.conio_screen_bank | VERA_INC_1;
217  *VERA_DATA0 = c;

224  cputln();
