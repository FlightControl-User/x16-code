5 struct MOS7360_TED {
6  // Counter #01. It always starts to decrement from the last written value into it.
7  unsigned int COUNTER1;
8  // Counter #02. It runs freely from $ffff.
9  unsigned int COUNTER2;
10  // Counter #03. Same as above.
11  unsigned int COUNTER3;
12  // Mostly the same as VIC's $d011.
13  // Bit 0,1,2 : Vertical smooth-scrolling
14  // Bit 3 : 24/25 rows screen
15  // Bit 4 : Blank screen
16  // Bit 5 : Bitplane mode
17  // Bit 6 : Enhanced color mode
18  // Bit 7 : TED's internal test, it should be 0.
19  unsigned char CONTROL1;
20  // Most similar VIC-reg is $d016.
21  // Bit 0,1,2 : Horizontal smooth-scrolling
22  // Bit 3 : 40/38 columns screen
23  // Bit 4 : Multicolor mode
24  // Bit 5 : TED stop. If set, the TED stops it's counters and screen-generating, only single clock and refresh
25  // cycles remain.
26  // Bit 6 : PAL/NTSC. 0:PAL, 1:NTSC
27  // Bit 7 : Disable reverse mode. If 0, we got 128 characters and higmost bit tells if the character should
28  // appear in inverse. If set, no inverse mode but 256 characters.
29  unsigned char CONTROL2;
30  // Keyboard input latch. Giving a strobe - writing to the register, the latch stores the values of the input-lines.
31  // Then, we can read them from this register.
32  volatile unsigned char KEYBOARD_INPUT;
33  // Interrupt request register. When a counter sends want to send an IRQ, it's bit will appear as a 0; then, if the
34  // IRQ was caused then highmost bit is set.
35  // Bit 0 : Unused
36  // Bit 1 : Raster-counter
37  // Bit 2 : Lightpen. Not implemented.
38  // Bit 3 : Counter #1
39  // Bit 4 : Counter #2
40  // Bit 5 : Unused
41  // Bit 6 : Counter #3
42  // Bit 7 : Interrupt occured. This bit is set when an IRQ was enabled and therefore, the IRQ was sent to the
43  // processor. Physically, this is the negated level of the TED's IRQ output. The IRQ should be
44  // deleted with writing the register-value back after accepting an interrupt.
45  unsigned char IRQ_REQUEST;
46  // Interrupt mask register. These bits could be used to disable and enable interrupt-sources. When a place is set to
47  // 1, that will be able to cause an interrupt to the processor. If not, the sign of the interrupt request will only
48  // be appear in the above register.
49  // Bit 0 : 9th bit of RASTER_IRQ (see there)
50  // Bit 1 : Raster-counter
51  // Bit 2 : Lightpen. Not implemented.
52  // Bit 3 : Counter #1
53  // Bit 4 : Counter #2
54  // Bit 5 : Unused
55  // Bit 6 : Counter #3
56  // Bit 7 : Unused
57  unsigned char IRQ_MASK;
58  // Raster interrupt register. Same as $d012 when writing; it stores the position of occuring raster interrupt.
59  // Higmost bit is in IRQ_REQUEST's 0. bit.
60  unsigned char RASTER_IRQ;
61  // Hardware-cursor position (10 bits). Lower bits: CURSOR_LO, higher 2 bits in CURSOR_HI's 0. and 1. places.
62  // Beyond 1000 the cursor is not seeable.
63  unsigned char CURSOR_HI;
64  unsigned char CURSOR_LO;
65  // First sound-source's frq-value's lowmost 8 bit. More 2 bits are in $ff10's 0. and 1. places.
66  unsigned char CH1_FREQ_LO;
67  // Second sound-source, lowmost 8 bits. More 2 bits in $ff12, 0. and 1. places.
68  // The sound register-value can be calculated as
69  // reg=1024-(111860.781/frq[Hz]) (NTSC)
70  // reg=1024-(111840.45 /frq[Hz]) (PAL)
71  unsigned char CH2_FREQ_LO;
72  // First sound-source, higmost 2 bits. 2-7 bits are unused.
73  unsigned char CH1_FREQ_HI;
74  // Sound control register.
75  // Bit 0-3 : Volume. Maximum value is 8.
76  // Bit 4 : Sound #1 on/off. (implicit squarewave)
77  // Bit 5 : Sound #2 squarewave on/off.
78  // Bit 6 : Sound #2 noise on/off. If You set both, the square will sound.
79  // Bit 7 : D/A mode. See above for more.
80  unsigned char SOUND_CONTROL;
81  // Bitmap Address and Miscellaneous Control
82  // Bit 0,1 : 2nd sound-source, highmost bits.
83  // Bit 2 : Character generator in ROM or RAM. When set, TED will enable ROM when trying to get data from the
84  // charactergenerator to build screen. Else, it will give out control-signals to the DRAM's.
85  // Bit 3,4,5 : These bits tell, where to find bitplane in the memory when using bitplane-mode. TED assumes them
86  // as A15,A14 and A13 bits. So, the bitplanes can be switched as 8K pages, anywhere in the 64K.
87  // Bit 6-7 : Unused.
88  unsigned char MEMORY1;
89  // Character Description Map Address and Miscellaneous Control
90  // Bit 0 : A sign to having control about memory paging. This bit always sets to 1 when ROM is active over
91  // $8000. Else, it will be 0. READ ONLY.
92  // Bit 1 : Force single clock mode. Then, TED will disable to generate twice clock.
93  // Bit 2-7 : Charactergenerator. Bit 7 corresponds to A15, 6 to A14 and so on. This value shows and sets the
94  // start of the charactergenerator. It can be paged as $400 bytes. Use with addition of CONTROL3-2.bit.
95  unsigned char MEMORY2;
96  // Screen Matrix address
97  // Bit 0-2 : Unused
98  // Bit 3-7 : Start of the video-ram. Bit 7 also corresponds to the A15 line as above. So, video-ram is mappable
99  // as $800 bytes - 2K. The above $ff12-2.bit doesn't affect this, but the actual RAM/ROM mapping
100  // (see at $ff3e/$ff3f and $ff13/0) does.
101  unsigned char MEMORY3;
102  // Background Color. Lower bits contain color-code, higher 3 luminance and highmost is ignored.
103  unsigned char BG_COLOR;
104  // Color register 1. Used in ECM and MCM modes.
105  unsigned char BG_COLOR1;
106  // Color register 2. Used in ECM and MCM modes.
107  unsigned char BG_COLOR2;
108  // Color register 3. Used in ECM and MCM modes.
109  unsigned char BG_COLOR3;
110  // Border Color. Lower bits contain color-code, higher 3 luminance and higmost is ignored.
111  unsigned char BORDER_COLOR;
112  // Actual character-position. TED counts the characters that it had fetched and put out to the screen.
113  // Lower bits: CHARPOS_LO, higher 2 bits in CHARPOS_HI's 0. and 1. places.
114  // The number is increasing by 40 after every characterline (8 rasterline).
115  unsigned char CHARPOS_HI;
116  unsigned char CHARPOS_LO;
117  // Actual position of vertical scanning. Higmost bit is in $ff1c. Read/Writeable!
118  unsigned char RASTER_HI;
119  unsigned char RASTER_LO;
120  // Actual position of horizontal scanning. R/W!. Lowmost bit is unused. It contains the TED's internal counter's
121  // highmost 8 bits. So, it increases 4 with every character. When writing, it seems to put the value to a
122  // functionally different register (writing back a reading value in right time affects the screen).
123  unsigned char HSCAN_POS;
124  // Bit 0,1,2 : Actual vertical scanning-line in a character-row. R/W!.
125  // Bit 3-6 : Flashing counter. It's value increases with every frame, and TED fits it's flashing feature to this
126  // register's reaching to 15.
127  // Bit 7 : Unused
128  unsigned char VSCAN_POS;
129  // Unused address space
130  unsigned char UNUSED[0x1d];
131  // Switching to ROM. A writing statement to this address will cause to turn on the ROM between $8000-$ffff. It's an
132  // other matter, which one; this time, only sure thing that it'll give CS signals instead of RAS', CAS' and MUX.
133  // See $ff13/0 and $ff14
134  unsigned char ROM_SWITCH;
135  // Switching to RAM. The opposite of ROM_SWITCH.
136  unsigned char RAM_SWITCH;
137 };
