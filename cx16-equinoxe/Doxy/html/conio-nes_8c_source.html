113 void cscroll() {
114  if(conio_cursor_y==CONIO_HEIGHT) {
115  if(conio_scroll_enable) {
116  // Scroll lines up
117  char* line1 = CONIO_SCREEN_TEXT;
118  char* line2 = CONIO_SCREEN_TEXT+CONIO_WIDTH;
119  for(char y=0;y<CONIO_HEIGHT-1;y++) {
120  ppuDataFetch(conio_cscroll_buffer, line2, CONIO_WIDTH);
121  ppuDataTransfer(line1, conio_cscroll_buffer, CONIO_WIDTH);
122  line1 += CONIO_WIDTH;

124  }

127  conio_line_text -= CONIO_WIDTH;
